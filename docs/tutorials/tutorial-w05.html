ota2009 <- read_csv("data/ota2009.csv") %>%
  filter(
    Procedure == "TrialProc",
    Contrast != "F"
  ) %>%
  mutate(
    Subject = as.factor(Subject),
    RT_log = log(Words.RT)
  )

ota2009_bm <- brm(
  RT_log ~
    Condition * Contrast +
    (Condition * Contrast | Subject),
  family = gaussian,
  backend = "cmdstanr",
  data = ota2009,
  cores = 4,
  seed = my_seed,
  file = "data/cache/ota2009_bm"
)
